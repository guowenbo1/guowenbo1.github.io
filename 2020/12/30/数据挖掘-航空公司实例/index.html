import numpy as np

import csv

写入xls函数

#function of write into csv
def writeXls(dataSet,dataName):

    #加newline = ''，否则会出现空行
    with open(dataName,'a',encoding='utf-8',newline='') as fo:
        writer = csv.writer(fo)
        for i in dataSet:
            writer.writerow(i)

    return 1

1、数据预处理

数据清理：给定原始数据air_data.csv, 去除原始数据中缺失值记录。处理方法如下：保留票价非零，或者平均折扣率与飞行公里数同时为0的记录。

index1=data[ ‘SUM_YR_1’] !=0

index2=data[ ‘SUM_YR_2’] !=0

index3=data[‘SEG_KM_SUM’] ==0 & data[‘AVG_DISCOUNT’]==0

data = data[ index1 | index2|index3]

【要求1】：将清理后的数据保存为data_cleaned.csv 或data_cleaned.xls

#function of data preprocess
def preProcess(dataSet,dataLabel):
    #useful label index
    sumYr1Col = int(np.where(dataLabel=='SUM_YR_1')[0])
    sumYr2Col = int(np.where(dataLabel=='SUM_YR_2')[0])
    segKmCol = int(np.where(dataLabel=='SEG_KM_SUM')[0])
    avgDisCol = int(np.where(dataLabel=='avg_discount')[0])

    dataClean = np.zeros([0, len(dataSet[0])])
    for i in range(len(dataSet)):
        if dataSet[i][sumYr1Col] != 0 and dataSet[i][sumYr2Col] != 0:
            # np.append指定axis的话需要两个数组的维度是一样的，所以用reshape强制改变dataset的维度为2，否则为1
            dataClean = np.append(dataClean, dataSet[i].reshape(1, len(dataSet[0])), axis=0)
        elif dataSet[i][segKmCol] == 0 and dataSet[i][avgDisCol] == 0:
            dataClean = np.append(dataClean, dataSet[i].reshape(1, len(dataSet[0])), axis=0)

    return dataClean

if __name__ =='__main__':
    dataPath = 'air_data.csv'

    #read label and dataset
    dataLabel = genfromtxt(dataPath, delimiter=',', dtype=str,encoding=None,max_rows=1)
    dataSet = genfromtxt(dataPath, delimiter=',', dtype=str,encoding='unicode_escape',max_rows=1000)

    #dataset preprocessed
    dataSet = preProcess(dataSet,dataLabel)

    #clean data write in xls file
    writeXls(dataSet,'data_cleaned.csv')

3、数据变换
采用z-core对上面数据进行标准化处理, 具体计算如下：
（1）L = LOAD_TIME-FFP_DATE
会员入会时间距观测窗口结束的月数=观测窗口的结束时间-入会时间（单位：月）
(2) R = LAST_TO_END
客户最近一次乘坐公司飞机距观测窗口结束的月数=最后一次乘机时间至观察窗口末端时长（单位：月）
(3) F = FLIGHT_COUNT
客户在观测窗口内乘坐公司飞机的次数=观测窗口的飞行次数（单位：次）
(4) M = SEG_KM_SUM
客户在观测时间内在公司累计的飞行里程=观测窗口的总飞行公里数（单位：公里）
(5) C = AVG_DISCOUNT
客户在观测时间内乘坐舱位所对应的折扣系数的平均值=平均折扣率（单位：无）
【要求3】：将变换后的数据保存为zscoreddata.xls

#read zscore data
    dataPath = 'zscoredata.csv'
    dataLabel = genfromtxt(dataPath, delimiter=',', dtype=str,encoding=None,max_rows=1)
    dataSet = genfromtxt(dataPath, delimiter=',', dtype=str,encoding=None,skip_header=1)

    #L = LOAD_TIME-FFP_DATE 会员入会时间距观测窗口结束的月数=观测窗口的结束时间-入会时间（单位：月)
    loadTimeCol = int(np.where(dataLabel == 'LOAD_TIME')[0])
    FFPCol = int(np.where(dataLabel == 'FFP_DATE')[0])
    # R = LAST_TO_END 客户最近一次乘坐公司飞机距观测窗口结束的月数 = 最后一次乘机时间至观察窗口末端时长（单位：月）
    lastEndCol = int(np.where(dataLabel == 'LAST_TO_END')[0])

    # F = FLIGHT_COUNT 客户在观测窗口内乘坐公司飞机的次数 = 观测窗口的飞行次数（单位：次）
    # 客户都是有过乘坐几千万次飞机经历的人？？？？？？？？？
    FlightCountCol = int(np.where(dataLabel == 'FLIGHT_COUNT')[0])

    # M = SEG_KM_SUM客户在观测时间内在公司累计的飞行里程 = 观测窗口的总飞行公里数（单位：公里）
    SegCol = int(np.where(dataLabel == 'SEG_KM_SUM')[0])
    # C = AVG_DISCOUNT 客户在观测时间内乘坐舱位所对应的折扣系数的平均值 = 平均折扣率（单位：无）
    avgDisCol = int(np.where(dataLabel == 'avg_discount')[0])


    #2014/03/31 换成可识别的日期2014-03-31
    dataSet[:, loadTimeCol] = np.char.replace(dataSet[:,loadTimeCol],"/","-")
    dataSet[:, FFPCol] = np.char.replace(dataSet[:,FFPCol],"/","-")

    a = dataSet[:,loadTimeCol].astype('datetime64')
    b = dataSet[:,FFPCol].astype('datetime64')
    l = (a - b).astype('float')

    # +1是为了加上标签行
    zscoredData = np.zeros([len(dataSet) + 1, 5],dtype='O')
    # 加标签行
    zscoredData[0,:] = ['l','R','M','F','C']
    # 加数据
    zscoredData[1:,0] = l
    zscoredData[1:,1] = dataSet[:,lastEndCol]
    zscoredData[1:,2] = dataSet[:,FlightCountCol]
    zscoredData[1:,3] = dataSet[:,SegCol]
    zscoredData[1:,4] = dataSet[:,avgDisCol]
    #写入文件
    writeXls(zscoredData,'zscoreddata.csv')
