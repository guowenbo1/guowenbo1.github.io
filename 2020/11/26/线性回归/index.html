线性回归

Eric Guo · 2020-11-26 · 次阅读

In this part of this exercise, you will implement linear regression with one variable to predict profits for a food truck. Suppose you are the CEO of a restaurant franchise and are considering different cities for opening a new outlet. The chain already has trucks in various cities and you have data for profits and populations from the cities. You would like to use this data to help you select which city to expand to next.
The file ex1data1.txt contains the dataset for our linear regression problem. The first column is the population of a city and the second column is the profit of a food truck in that city. A negative value for profit indicates a loss. 


import matplotlib.pyplot as plt
from sklearn import linear_model
import numpy as np


# MMSE(mean square error)均方误差
# theat0参数 theat1参数 xList自变量 yList因变量
def MSE(theat0 ,theat1 ,xList ,yList):
    sum = 0
    m = len(xList)
    for i in range(m):
        sum += (theat0 + theat1 * xList[i] - yList[i]) ** 2
    value = sum / ( 2 *m)
    return round(value ,10)


# 梯度下降算法迭代更新theat0和theat1
def gradient_descent(theat0 ,theat1 ,alpha ,xList ,yList):
    m = len(xList)
    sum0 = 0
    sum1 = 0
    for i in range(m):
        sum0 += theat0 +theat1 * xList[i] - yList[i]
        sum1 += (theat0 +theat1 * xList[i] - yList[i]) * xList[i]
    theat0 = theat0 - alpha * sum0 / m
    theat1 = theat1 - alpha * sum1 / m
    return round(theat0 ,10) ,round(theat1 ,10)



if __name__ == "__main__":
    file_input_address = "chap3/chap3/作业/ex1data1.txt"  # 相对路径

    with open(file_input_address) as fo:
        b = fo.readlines()  # 返回值为list 每个值由每行构成的字符组成 type(b(0)) = str
        population = []  # 存储人口的列表
        profit = []  # 存储利润的列表
        for i in b:
            i_array = i.strip("\n").split(",")  # 去除\n并以，分割字符串
            population.append(float(i_array[0]))  # 将第一个值赋给population
            profit.append(float(i_array[1]))  # 将第二个值赋给profit
            i_array.clear()

    theat0 = 0  # 第一个参数
    theat1 = 0  # 第二个参数
    alpha = 0.01  # 学习率
    epslion = 0.0001  # 最小误差值 当两次迭代的MSE小于此值时停止迭代
    error0 = 0
    count = 0
    error = []
    while True:
        count += 1
        theat0 ,theat1 = gradient_descent(theat0 ,theat1 ,alpha ,population ,profit)
        error1 = MSE(theat0 ,theat1 ,population ,profit) #计算迭代的误差
        error.append(error1)
        if abs(error1 - error0) <=epslion:
            break
        else:
            error0 = error1

    print("theat0 is" ,theat0 ," theat1 is" ,theat1)
    print("迭代次数为" ,count)

    #转换成nparray才可用sklearn
    population = np.array(population).reshape(-1,1)
    profit = np.array(profit).reshape(-1,1)
    yy = theat0 + theat1*population

    #绘制损失函数的变化
    plt.plot(error) #x值缺省时默认为len(error)
    plt.ylabel('Cost J')
    plt.xlabel('Iterations');
    plt.show()

    ## visualizing the data
    reg = linear_model.LinearRegression()
    reg.fit(population,profit)
    predicted = reg.predict(population)

    ## visualizing the data
    # 绘制散点图 参数：x横轴 y纵轴
    plt.scatter(population, profit, marker='x')
    plt.plot(population,yy,'r-',label = "my model") #我的预测函数
    plt.plot(population,predicted,'g-',label = 'sklearn model') #sklearn的预测函数

    # 绘制x轴和y轴坐标
    plt.xlabel("population")
    plt.ylabel("profit")

    # 显示图形
    plt.show()
