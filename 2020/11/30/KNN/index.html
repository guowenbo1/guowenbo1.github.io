Eric Guo · 2020-11-30 · 次阅读

KNN

数据集：Iris也称鸢尾花卉数据集，是一类多重变量分析的数据集。该数据集是4个最流行的机器学习数据集之一。通过花萼长度，花萼宽度，花瓣长度，花瓣宽度4个属性预测鸢尾花卉属于（Setosa，Versicolour，Virginica）三个种类中的哪一类。

实验要求：

1、 读入Iris.csv数据，并将数据的70%做为训练数据，30%做为测试数据；

2、 根据KNN算法思想，实现KNN算法，并使用训练数据训练KNN分类模型；

3、 使用剩余30%数据对KNN分类模型进行测试，并计算出分类精度。（可考虑使用可视化界面进行分类效果分析）

4、 可以直接用sklearn库中的neighbors.KNeighborsClassifier()

import matplotlib.pyplot as plt
import numpy as np
from numpy import genfromtxt
from sklearn.metrics import confusion_matrix

# knn的实现
# trainData为训练集；testData为检验集（不知道标签的集）；labels为训练集的标签；k为要计算欧氏距离的元素的个数
def knn(trainData, testData, labels, k):
    # 计算训练样本的行数
    row = trainData.shape[0]

    # 计算训练样本和测试样本的差值
    diff = np.tile(testData, (row, 1)) - trainData
    #np.tile(testData, (rowSize, 1)) 是将 testData 这个数据扩展为 rowSize 列，这样能避免运算错误；

    # 计算差值的平方和
    sqrDiff = diff ** 2
    sqrDiffSum = sqrDiff.sum(axis=1)  #求和

    # 计算距离（欧式距离）
    distances = sqrDiffSum ** 0.5

    # 对所得的距离从低到高进行排序
    sortDistance = distances.argsort()  #argsort函数返回的是数组值从小到大的索引值;返还值类型为NdArray
    #例：若argsort返还值为[91,72,79]意思为distance[91]最小（最近），其次是distanc[72】,之后是distance[79]
    #值越小意味着越近

    # print('sotdistance is',sortDistance)
    # print('sortdistance[0] is',sortDistance[0])

    #字典：键为标签名，值为此标签名出现的次数
    count = {}
    for i in range(k):
        vote = labels[sortDistance[i]] #前三个最小的距离的索引值；labels和distance的相同的索引值代表着相同的数据
        count[vote] = count.get(vote, 0) + 1  #dic.get()返还指定键的值

    # 对类别出现的频数从高到低进行排序
    sortCount = sorted(count.items(), key=lambda d:d[1], reverse=True)
    #sorted(count.items(), key=lambda d:d[1], reverse=True) 排序函数，里面的参数 key=lambda d:d[1],
    #reverse=True 表示按照 count 这个字典的值（value）从高到低排序，
    #如果把 1 换成 0，则是按字典的键（key）从高到低排序。把 True 换成 False 则是从低到高排序。

    # 返回出现频数最高的类别
    return sortCount[0][0]


#把数据集分成训练和检验两个样本；dataset为要分割的数据集，count为训练集的百分比
def DataSplit(data,count):
    dataCount = data.shape[0] - 1  # 数据的行数; -1是为了把第一行的属性减去
    testCount = round(dataCount * count)  # 要设置的训练集的行数

    # 分割测试和检验数据集
    dataTest = data[0:testCount:]  # 训练数据集的x值
    dataVerify = data[testCount:]  # 检验数据集的x值

    return dataTest,dataVerify


#计算准确度；输入为两个标签，计算差异度
def CalAccuracy(x,y):
    correctCount = 0
    for i in range(x.shape[0]):
        if y[i] == x[i]:
            correctCount += 1
    accuracy = correctCount / x.shape[0]
    return accuracy


# #混淆矩阵的构建
def CmPlot(trainLabel,testLabel):

    #confusion_matrix输入真实和预测的标签列表，返还个矩阵
    cm = confusion_matrix(trainLabel,testLabel)

    #matshow函数绘制成矩阵图像
    plt.matshow(cm, cmap=plt.cm.Greens)

    #colorbar添加渐变色条
    plt.colorbar()

    #矩阵上显示数字
    for x in range(len(cm)):
        for y in range(len(cm)):
            plt.annotate(cm[x, y], xy=(
                x, y), horizontalalignment='center', verticalalignment='center')

    #中文显示不出来
    plt.ylabel('true')
    plt.xlabel('predict')

    return plt


if __name__ == '__main__':
    dataPath = 'iris1.csv'
    dataSet = genfromtxt(dataPath, delimiter=',',dtype=None,encoding='utf-8')
    label = dataSet[1:,-1]   #最后一行的数据 即标签列
    data = dataSet[1:,0:-1]  #除了标签列的数据
    trainData,testData = DataSplit(data,0.7)    #以70%为界限去分割训练集和数据集
    trainData = trainData.astype(np.float)      #将非label标签的数据转换成float类型，方便后续欧氏距离计算
    testData = testData.astype(np.float)
    trainLabel,testLabel = DataSplit(label,0.7)

    y = [] #y存储根据根据knn模型所得到测试集数据的分类 的列表
    for i in testData:
        value = knn(trainData,i,trainLabel,3)
        y.append(value)
    y = np.array(y)

    #计算准确度
    accuracy = CalAccuracy(y,testLabel)
    print('分类精度为',accuracy)

    #混淆矩阵的对角线区域为TP(true positive);及预测结果与真实值一致的区域
    CmPlot(testLabel,y).show()

实验：基于不同的算法算欧氏距离去实现KNN

import numpy as np
from numpy import genfromtxt
import time

#利用np里的linalg去计算二范数（欧氏距离）
#返还矩阵的下标i，k即为第I个元素与第K个元素的欧氏距离
def disNorm(dataSet):
    disMatrix = np.zeros([len(dataSet),len(dataSet)]) #生成0矩阵
    for k in range(len(dataSet)):
        for j in range(len(dataSet)):
            # if j != k:
            if j > k:
                distance = np.linalg.norm(dataSet[k] - dataSet[j])
                # print('数据里的第(%d)个元素与第(%d)个元素的欧氏距离为（%f)' % (k,j,distance))
                disMatrix[k][j] = distance
    return disMatrix


#利用scipy库里的pdist函数去计算欧几里得距离
#返还矩阵的下标i，k即为第I个元素与第K个元素的欧氏距离
def disPdist(dataSet):
    disMatrix = np.zeros([len(dataSet), len(dataSet)])
    from scipy.spatial.distance import pdist
    eucDis = pdist(dataSet)  #pdist返还值为ndarray类型
    sum = 0
    for k in range(len(dataSet) - 1):
        for j in range(len(dataSet) - 1):
            if j >= k:
                a = sum + (j - k)
                disMatrix[k][j+1] = eucDis[a] #j+1是因为 eucDis是没有自己和自己的距离，而每行的对角线的那个点是自己对自己的距离
        sum = sum + len(dataSet) - k -1
    return disMatrix


#基于np.norm的KNN
def knnNorm(trainData, testData,labels,k):
    trainData = np.vstack((trainData,testData))

    ###基于np。norm
    trainData = disPdist(trainData)

    ####key重复了
    # disDic = {} #字典 key是label； value是距离
    # for i in range(0,len(trainData)-1):
    #     dis = float(trainData[i:i+1,-1])
    #     label = labels[i]
    #     disDic[label] = dis

    distance = np.zeros([len(trainData) - 1, 2], dtype='O')  # dtype为float和complex；None 都会出错
    for i in range(0, len(trainData) - 1):  # 因为多加了个tesddata
        dis = float(trainData[i:i + 1, -1])
        label = labels[i]
        distance[i] = (label, dis)

    # 对所得的距离从低到高进行排序
    sortDistance = distance[:,-1].argsort()   ####-1去掉
    #argsort函数返回的是数组值从小到大的索引值;返还值类型为NdArray
    #例：若argsort返还值为[91,72,79]意思为distance[91]最小（最近），其次是distanc[72】,之后是distance[79]
    #值越小意味着越近


    #字典：键为标签名，值为此标签名出现的次数
    count = {}
    for i in range(k): #若k为3 循环为0,1,2
        vote = labels[sortDistance[i]] #前三个最小的距离的索引值；labels和distance的相同的索引值代表着相同的数据
        count[vote] = count.get(vote, 0) + 1  #dic.get()返还指定键的值

    # 对类别出现的频数从高到低进行排序
    sortCount = sorted(count.items(), key=lambda d:d[1], reverse=True)
    #sorted(count.items(), key=lambda d:d[1], reverse=True) 排序函数，里面的参数 key=lambda d:d[1],
    #reverse=True 表示按照 count 这个字典的值（value）从高到低排序，
    #如果把 1 换成 0，则是按字典的键（key）从高到低排序。把 True 换成 False 则是从低到高排序。

    # 返回出现频数最高的类别
    return sortCount[0][0]
